// ── Helper: set field state ────────────────────────────
function setField(inputEl, iconEl, hintEl, valid, msg) {
  inputEl.className = 'form-input ' + (valid ? 'valid' : 'invalid');
  iconEl.textContent = valid ? '✓' : '✕';
  iconEl.className   = 'input-icon show';
  iconEl.style.color = valid ? '#0F6E56' : '#B91C1C';
  hintEl.textContent = msg;
  hintEl.className   = 'hint ' + (valid ? 'success' : 'error');
}

// ── Username validation ─────────────────────────────────
function validateUsername(value) {
  value = value.trim();
  if (!value)           return { ok: false, msg: 'Username cannot be empty' };
  if (value.length < 3)  return { ok: false, msg: 'At least 3 characters required' };
  if (value.length > 16) return { ok: false, msg: 'Maximum 16 characters allowed' };
  if (!/^[a-zA-Z0-9]+$/.test(value))
                         return { ok: false, msg: 'Letters and numbers only' };
  return { ok: true, msg: 'Looks good!' };
}

// ── Email validation ────────────────────────────────────
function validateEmail(value) {
  value = value.trim();
  if (!value) return { ok: false, msg: 'Email cannot be empty' };
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value))
              return { ok: false, msg: 'Enter a valid email address' };
  return { ok: true, msg: 'Valid email!' };
}

// ── Password validation ─────────────────────────────────
function validatePassword(value) {
  if (!value)              return { ok: false, msg: 'Password cannot be empty' };
  if (value.length < 8)   return { ok: false, msg: 'At least 8 characters required' };
  if (!/[A-Z]/.test(value)) return { ok: false, msg: 'Add at least one uppercase letter' };
  if (!/[0-9]/.test(value)) return { ok: false, msg: 'Add at least one number' };
  if (!/[^A-Za-z0-9]/.test(value))
                           return { ok: false, msg: 'Add at least one special character' };
  return { ok: true, msg: 'Strong password!' };
}

// ── Wire up event listeners ─────────────────────────────
document.getElementById('usernameInput')
  .addEventListener('input', function() {
    const result = validateUsername(this.value);
    setField(this, usernameIcon, usernameHint, result.ok, result.msg);
    checkAll();
  });

document.getElementById('submitBtn')
